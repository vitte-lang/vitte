Commented Keyword Gallery

This gallery shows the current keywords in compact examples with comments so the whole surface can be read quickly from one page.

Use it as the companion to the individual keyword sheets: each block below highlights a cluster of keywords in context.

Declarations

// `space`, `pull`, `use`, and `share` shape the module boundary.
space demo/app
pull std/io
use std/core as core
share demo/api

// `const`, `type`, `form`, `field`, and `trait` describe the public shape.
const DEBUG: bool = true
type Name = string
form User {
  field name as string
}
trait Printable {}

Core Execution

// `proc` and `entry` define the executable surface.
proc main() -> int {
  // `let` introduces a binding, `make` initializes a named value.
  let count: int = 0
  make total = 1

  // `set` updates the value, `emit` writes a visible side effect.
  set total = total + 1
  emit "ready"

  // `give` returns from the current block, while `return` ends the outer flow.
  give total
}

entry app at demo/app {
  return 0
}

Control Flow

// `if` and `else` are the basic decision pair.
if DEBUG {
  give 1
} else {
  give 0
}

// `loop`, `for`, `in`, `break`, and `continue` explain repetition clearly.
loop {
  for item in items {
    if item == 0 {
      break
    }
    continue
  }
}

Matching

// `pick` and `case` define a closed value set.
pick Result {
  case Ok,
  case Err
}

// `match`, `select`, `when`, `is`, and `otherwise` split the decision tree.
match result {
  case Ok { give 0 }
  otherwise { give 1 }
}

select value {
  when 0 { give 0 }
  when 1 { give 1 }
  is 2 { give 2 }
  otherwise { give 3 }
}

Logic And Operators

// `and`, `or`, and `not` are read as boolean logic.
if ready and not failed or debug {
  give 1
}

// `as` and `at` stay visible in imports and entry points.
use demo/text as text
entry main at demo/app {
  return 0
}

// `all` is the whole-set keyword; keep its meaning visible in comments.
# all: read the corresponding `all.html` sheet for the dedicated contract.

Advanced Surface

// `macro` can rewrite structured fragments.
macro trace(x) {
  emit x
}

// `unsafe` marks an explicit escape hatch, `asm` embeds raw machine intent.
unsafe {
  asm "nop"
}

// `bool`, `string`, `int`, `true`, and `false` stay visible in the same reading block.
const enabled: bool = false
const banner: string = "Vitte"
const answer: int = 42
