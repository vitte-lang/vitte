vitte 0.1.0
usage: vitte <command> [options] [file]

core commands:
  check FILE        validate a source file
  build FILE -o OUT build a native executable
  run FILE          build and run a source file
  parse FILE        parse only

package commands:
  pkg check         validate packages under src/vitte/packages
  pkg matrix        run package dependency/import checks
  pkg selftest      run package package checks and matrix

debug commands:
  dump-tokens FILE
  dump-ast FILE
  dump-hir FILE
  dump-mir FILE
  dump-native-ir --src FILE
  selfhost-source

common flags:
  --src PATH  --out PATH  -o PATH  --trace-pipeline  --strict
  --dump-ast-json  --dump-hir-json  --dump-mir-json  --diagnostics-json
